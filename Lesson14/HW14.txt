1. /*Для своей базы данных создайте объект для нахождения общего рейтинга топ 50 продавцов за всю историю продаж,
где рейтинг определяется количеством баллов за количество продаж в день (10 и более продаж – 3 балла, 5-10 – 2 балла, меньше 5 – 1 балл).*/

CREATE VIEW TOP50Vendor
AS
SELECT TOP (50) WITH TIES FirstName, LastName, SUM(RatingManagers) AS FinalRating
FROM
(
SELECT M.IDManager 
	,M.FirstName
	,M.LastName
	,S.DataSales
	,COUNT(M.IDManager) AS cnt_manager
	,CASE 
	WHEN COUNT(M.IDManager) > 10 THEN 3
	WHEN COUNT(M.IDManager) BETWEEN 5 AND 10 THEN 2
	ELSE 1
	END AS RatingManagers
FROM [dbo].[ManagersDim] M 
LEFT JOIN [dbo].[SalesFct] S ON S.IDManager = M.IDManager
GROUP BY M.IDManager ,M.FirstName, M.LastName, S.DataSales
) t
GROUP BY FirstName, LastName
ORDER BY FinalRating DESC

SELECT * FROM TOP50Vendor
